local function joinTeam(teamName, position)
if teamName == "None" or position == "None" then return end

local team = teamsService:FindFirstChild(teamName)

    local args = {  
        [1] = {  
            ["Team"] = team,  
            ["TeamPosition"] = position  
        }  
    }  
    pcall(function()  
        gamemodeComm.RE._RequestJoin:FireServer(unpack(args))  
    end)  
end

end

local function getGoalPosition()
local stadium = Workspace:FindFirstChild("Stadium")
if stadium and stadium:FindFirstChild("Teams") then
local goal = stadium.Teams:FindFirstChild(TargetGoal) and stadium.Teams[TargetGoal]:FindFirstChild("Goal")
if goal and goal:FindFirstChild("Hitbox") then return goal.Hitbox.Position end
end
return Vector3.new(0, 0, 0)
end

RunService.Heartbeat:Connect(function()
if AutoFarmEnabled or AutoScoreEnabled then
local ball = getBall()
local char = LocalPlayer.Character
local hrp = char and char:FindFirstChild("HumanoidRootPart")
local goalPos = getGoalPosition()

if ball and hrp and goalPos ~= Vector3.new(0,0,0) then  
        local dist = (hrp.Position - ball.Position).Magnitude  
        if AutoFarmEnabled and dist > 5 then  
            hrp.CFrame = CFrame.lookAt(ball.Position + Vector3.new(0, 0, 3), ball.Position)  
        end  
        if dist <= 7 then  
            hrp.CFrame = CFrame.lookAt(hrp.Position, goalPos)  
            hrp.Velocity = hrp.CFrame.LookVector * 45  
        end  
    end  
end

end)

local LastBallHit = tick()
RunService.RenderStepped:Connect(function()
local char = LocalPlayer.Character
local hrp = char and char:FindFirstChild("HumanoidRootPart")
local phitbox = char and char:FindFirstChild("Hitbox")
local ball = getBall()

if phitbox and phitbox:IsA("BasePart") then  
    if not originalHitboxSize then originalHitboxSize = phitbox.Size end  
    if HitboxEnabled then  
        phitbox.Size = Vector3.new(HitboxSize, HitboxSize, HitboxSize)  
        phitbox.Transparency = 0.6; phitbox.Color = Color3.fromRGB(0, 255, 100); phitbox.CanCollide = false  
        if ball and hrp and (hrp.Position - ball.Position).Magnitude <= HitboxSize then  
            ball.AssemblyLinearVelocity = (hrp.Position - ball.Position).Unit * 5  
        end  
    else  
        phitbox.Size = originalHitboxSize; phitbox.Transparency = 1  
    end  
end  

if ball and BallVelocityEnabled and hrp then  
    if (hrp.Position - ball.Position).Magnitude < 15 and ball.AssemblyLinearVelocity.Magnitude > 10 and tick() - LastBallHit > 0.5 then  
        ball.AssemblyLinearVelocity = ball.AssemblyLinearVelocity * BallVelocityMultiplier  
        LastBallHit = tick()  
    end  
end

end)

local LeaderboardText = TabLeader:CreateParagraph({Title = "Top Players", Content = "Fetching data..."})

local function updateLeaderboard()
local statsStr = ""
local playerList = Players:GetPlayers()
table.sort(playerList, function(a, b)
local aG = (a:FindFirstChild("leaderstats") and a.leaderstats:FindFirstChild("Goals")) and a.leaderstats.Goals.Value or 0
local bG = (b:FindFirstChild("leaderstats") and b.leaderstats:FindFirstChild("Goals")) and b.leaderstats.Goals.Value or 0
return aG > bG
end)
for i, plr in ipairs(playerList) do
local g = (plr:FindFirstChild("leaderstats") and plr.leaderstats:FindFirstChild("Goals")) and plr.leaderstats.Goals.Value or 0
local t = plr.Team and plr.Team.Name or "No Team"
statsStr = statsStr .. i .. ". " .. plr.Name .. " | Team: " .. t .. " | Goals: " .. g .. "\n"
end
LeaderboardText:Set({Title = "Top Players (" .. #playerList .. " In Game)", Content = statsStr})
end

TabLeader:CreateButton({Name = "Refresh", Callback = updateLeaderboard})
task.spawn(function() while task.wait(5) do pcall(updateLeaderboard) end end)
