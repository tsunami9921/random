local function getGoalPosition()
    local stadium = Workspace:FindFirstChild("Stadium")
    if stadium and stadium:FindFirstChild("Teams") then
        local goal = stadium.Teams:FindFirstChild(TargetGoal) and stadium.Teams[TargetGoal]:FindFirstChild("Goal")
        if goal and goal:FindFirstChild("Hitbox") then return goal.Hitbox.Position end
    end
    return Vector3.new(0, 0, 0)
end

RunService.Heartbeat:Connect(function()
    if AutoFarmEnabled or AutoScoreEnabled then
        local ball = getBall()
        local char = LocalPlayer.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local goalPos = getGoalPosition()

        if ball and hrp and goalPos ~= Vector3.new(0,0,0) then
            local dist = (hrp.Position - ball.Position).Magnitude
            if AutoFarmEnabled and dist > 5 then
                hrp.CFrame = CFrame.lookAt(ball.Position + Vector3.new(0, 0, 3), ball.Position)
            end
            if dist <= 7 then
                hrp.CFrame = CFrame.lookAt(hrp.Position, goalPos)
                hrp.Velocity = hrp.CFrame.LookVector * 45
            end
        end
    end
end)

local HitboxEnabled, HitboxSize, BallVelocityEnabled, BallVelocityMultiplier = false, 5, false, 2
local originalHitboxSize = nil

TabBall:CreateToggle({Name = "📦 Enhanced Hitbox (Magnet)", CurrentValue = false, Callback = function(v) HitboxEnabled = v end})

local LastBallHit = tick()
RunService.RenderStepped:Connect(function()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    local phitbox = char and char:FindFirstChild("Hitbox")
    local ball = getBall()
    
    if phitbox and phitbox:IsA("BasePart") then
        if not originalHitboxSize then originalHitboxSize = phitbox.Size end
        if HitboxEnabled then
            phitbox.Size = Vector3.new(HitboxSize, HitboxSize, HitboxSize)
            phitbox.Transparency = 0.6; phitbox.Color = Color3.fromRGB(0, 255, 100); phitbox.CanCollide = false
            if ball and hrp and (hrp.Position - ball.Position).Magnitude <= HitboxSize then
                ball.AssemblyLinearVelocity = (hrp.Position - ball.Position).Unit * 5
            end
        else
            phitbox.Size = originalHitboxSize; phitbox.Transparency = 1
        end
    end

    if ball and BallVelocityEnabled and hrp then
        if (hrp.Position - ball.Position).Magnitude < 15 and ball.AssemblyLinearVelocity.Magnitude > 10 and tick() - LastBallHit > 0.5 then
            ball.AssemblyLinearVelocity = ball.AssemblyLinearVelocity * BallVelocityMultiplier
            LastBallHit = tick()
        end
    end
end)

-- Dribble Fix Engine
local function setupAnimatorFix(char)
    local hum = char:WaitForChild("Humanoid", 5)
    local animator = hum and hum:WaitForChild("Animator", 5)
    local hrp = char:WaitForChild("HumanoidRootPart", 5)
    if animator and hrp then
        animator.AnimationPlayed:Connect(function(track)
            if DribbleFixEnabled and track.Animation then
                local n = track.Animation.Name:lower()
                if n:find("dribble") or n:find("flick") or n:find("turn") or n:find("step") then
                    track:Stop()
                    local customAnim = Instance.new("Animation")
                    customAnim.AnimationId = frontFlipId
                    local newTrack = animator:LoadAnimation(customAnim)
                    newTrack:Play()
                    hrp.AssemblyLinearVelocity = hrp.CFrame.LookVector * 45 + Vector3.new(0, 15, 0)
                end
            end
        end)
    end
end
if LocalPlayer.Character then setupAnimatorFix(LocalPlayer.Character) end
LocalPlayer.CharacterAdded:Connect(setupAnimatorFix)

-- Client Skin Engine
local lastBall, lastSkin = nil, ""
RunService.Heartbeat:Connect(function()
    local ball = getBall()
    if ball and ball:IsA("BasePart") then
        if SkinEnabled and SelectedSkin ~= "None" then
            if ball ~= lastBall or SelectedSkin ~= lastSkin then
                if lastBall and lastBall:FindFirstChild("SLS_ClientSkin") then lastBall.SLS_ClientSkin:Destroy() end
                lastBall = ball; lastSkin = SelectedSkin
                pcall(function()
                    local targetTheme = ReplicatedStorage.Assets.Items.BallThemes:FindFirstChild(SelectedSkin, true)
                    if targetTheme then
                        local clone = targetTheme:Clone()
                        clone.Name = "SLS_ClientSkin"
                        for _, v in ipairs(clone:GetDescendants()) do if v:IsA("BasePart") then v.CanCollide=false; v.Anchored=false; v.Massless=true; v.CanTouch=false end end
                        if clone:IsA("BasePart") then clone.CanCollide=false; clone.Anchored=false; clone.Massless=true; clone.CanTouch=false end
                        clone.Parent = ball
                        local weldPart = clone:IsA("Model") and clone.PrimaryPart or clone
                        if clone:IsA("Model") then clone:PivotTo(ball.CFrame) else clone.CFrame = ball.CFrame end
                        local w = Instance.new("WeldConstraint"); w.Part0 = ball; w.Part1 = weldPart; w.Parent = clone
                    end
                end)
            end
            ball.Transparency = 1
            local cSkin = ball:FindFirstChild("SLS_ClientSkin")
            if cSkin then for _, v in ipairs(cSkin:GetDescendants()) do if v:IsA("BasePart") and v.Transparency>0.1 then v.Transparency=0 end end end
        else
            if lastBall and lastBall:FindFirstChild("SLS_ClientSkin") then lastBall.SLS_ClientSkin:Destroy(); lastBall.Transparency=0; lastSkin="" end
        end
    end
end)

-- ==================== LEADERBOARD ====================
local LeaderboardText = TabLeader:CreateParagraph({Title = "🏆 Top Players", Content = "Fetching data..."})

local function updateLeaderboard()
    local statsStr = ""
    local playerList = Players:GetPlayers()
    table.sort(playerList, function(a, b)
        local aG = (a:FindFirstChild("leaderstats") and a.leaderstats:FindFirstChild("Goals")) and a.leaderstats.Goals.Value or 0
        local bG = (b:FindFirstChild("leaderstats") and b.leaderstats:FindFirstChild("Goals")) and b.leaderstats.Goals.Value or 0
        return aG > bG
    end)
    for i, plr in ipairs(playerList) do
        local g = (plr:FindFirstChild("leaderstats") and plr.leaderstats:FindFirstChild("Goals")) and plr.leaderstats.Goals.Value or 0
        local t = plr.Team and plr.Team.Name or "No Team"
        statsStr = statsStr .. i .. ". " .. plr.Name .. " | Team: " .. t .. " | Goals: " .. g .. "\n"
    end
    LeaderboardText:Set({Title = "🏆 Top Players (" .. #playerList .. " In Game)", Content = statsStr})
end

TabLeader:CreateButton({Name = "🔄 Force Refresh", Callback = updateLeaderboard})
task.spawn(function() while task.wait(5) do pcall(updateLeaderboard) end end)
